          # 注意事项

          如果你使用的是 vscode 编辑器，在 vuecli 项目中，eslint 和 prettier 在保存的时候没有自动格式化，记得加上 vscode 的配置文件

          {
    // 自动保存,并进行 eslint 的自动格式化
    "editor.formatOnSave": true,
    "editor.codeActionsOnSave": {
        "source.fixAll.eslint": "explicit"
    },
    // 字体、 菜单栏字体
    // "editor.fontSize": 14,
    // "window.zoomLevel": -0.8,
    // "window.zoomLevel": -0.4,
    //
    "files.eol": "\n",
    "eslint.validate": [
        "javascript",
        "vue",
        "html",
        "json"
    ],
    // sass
    "easysass.formats": [
        {
            "format": "expanded",
            "extension": ".css"
        },
        {
            "format": "compressed",
            "extension": ".min.css"
        }
    ],
}
